# config/nginx/nginx.conf
events {
    worker_connections 1024;
}

http {
    include       /etc/nginx/mime.types;
    default_type  application/octet-stream;

    # Logging
    access_log /var/log/nginx/access.log;
    error_log /var/log/nginx/error.log;

    # Gzip compression
    gzip on;
    gzip_vary on;
    gzip_min_length 1024;
    gzip_types text/plain text/css application/json application/javascript text/xml application/xml application/xml+rss text/javascript;

    # Rate limiting
    limit_req_zone $binary_remote_addr zone=api:10m rate=10r/s;
    limit_req_zone $binary_remote_addr zone=web:10m rate=5r/s;

    # Upstream servers
    upstream backend {
        server backend:8000;
        keepalive 32;
    }

    upstream ai_service {
        server ai-service:8001;
        keepalive 16;
    }

    upstream frontend_dev {
        server frontend:3000;
        keepalive 16;
    }

    # Main server
    server {
        listen 80;
        server_name localhost shout-aloud.local;

        # Security headers
        add_header X-Frame-Options DENY;
        add_header X-Content-Type-Options nosniff;
        add_header X-XSS-Protection "1; mode=block";
        add_header Referrer-Policy strict-origin-when-cross-origin;

        # Frontend (development mode)
        location / {
            proxy_pass http://frontend_dev;
            proxy_http_version 1.1;
            proxy_set_header Upgrade $http_upgrade;
            proxy_set_header Connection 'upgrade';
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
            proxy_cache_bypass $http_upgrade;
        }

        # API endpoints
        location /api/ {
            limit_req zone=api burst=20 nodelay;
            
            proxy_pass http://backend/;
            proxy_http_version 1.1;
            proxy_set_header Connection "";
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
            
            # CORS headers
            add_header Access-Control-Allow-Origin *;
            add_header Access-Control-Allow-Methods "GET, POST, PUT, DELETE, OPTIONS";
            add_header Access-Control-Allow-Headers "Content-Type, Authorization";
            
            # Handle preflight
            if ($request_method = 'OPTIONS') {
                return 204;
            }
        }

        # AI service endpoints
        location /ai/ {
            limit_req zone=api burst=5 nodelay;
            
            proxy_pass http://ai_service/;
            proxy_http_version 1.1;
            proxy_set_header Connection "";
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
            
            # Increase timeout for AI processing
            proxy_connect_timeout 60s;
            proxy_send_timeout 300s;
            proxy_read_timeout 300s;
        }

        # Health check endpoint
        location /health {
            access_log off;
            return 200 "healthy\n";
            add_header Content-Type text/plain;
        }

        # IPFS gateway proxy
        location /ipfs/ {
            proxy_pass http://ipfs:8080/ipfs/;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        }

        # Monitoring endpoints
        location /metrics {
            proxy_pass http://prometheus:9090/metrics;
            allow 172.20.0.0/16;  # Only allow internal network
            deny all;
        }
    }
}

# database/init.sql
-- Initialize Shout Aloud database

-- Create users table
CREATE TABLE IF NOT EXISTS users (
    id SERIAL PRIMARY KEY,
    did VARCHAR(255) UNIQUE NOT NULL,
    identity_commitment VARCHAR(255) UNIQUE NOT NULL,
    municipality_code INTEGER,
    state_code INTEGER,
    country_code VARCHAR(3) DEFAULT 'MX',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Create proposals table
CREATE TABLE IF NOT EXISTS proposals (
    id SERIAL PRIMARY KEY,
    title VARCHAR(500) NOT NULL,
    summary TEXT NOT NULL,
    content TEXT NOT NULL,
    category VARCHAR(100) NOT NULL,
    scope VARCHAR(50) NOT NULL, -- 'municipal', 'state', 'federal'
    municipality_code INTEGER,
    state_code INTEGER,
    country_code VARCHAR(3) DEFAULT 'MX',
    status VARCHAR(50) DEFAULT 'active', -- 'active', 'closed', 'draft'
    source_url TEXT,
    author VARCHAR(255),
    deadline TIMESTAMP,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Create votes table
CREATE TABLE IF NOT EXISTS votes (
    id SERIAL PRIMARY KEY,
    proposal_id INTEGER REFERENCES proposals(id),
    nullifier_hash VARCHAR(255) UNIQUE NOT NULL,
    vote_commitment VARCHAR(255) NOT NULL,
    municipality_code INTEGER,
    state_code INTEGER,
    country_code VARCHAR(3) DEFAULT 'MX',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    CONSTRAINT unique_vote_per_proposal_per_user UNIQUE (proposal_id, nullifier_hash)
);

-- Create officials table for governance ratings
CREATE TABLE IF NOT EXISTS officials (
    id SERIAL PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    position VARCHAR(255) NOT NULL,
    level VARCHAR(50) NOT NULL, -- 'municipal', 'state', 'federal'
    municipality_code INTEGER,
    state_code INTEGER,
    country_code VARCHAR(3) DEFAULT 'MX',
    party VARCHAR(100),
    start_date DATE,
    end_date DATE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Create tags table for official ratings
CREATE TABLE IF NOT EXISTS tags (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) UNIQUE NOT NULL,
    category VARCHAR(50) NOT NULL, -- 'positive', 'negative', 'neutral'
    weight DECIMAL(3,2) DEFAULT 1.0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Create ratings table
CREATE TABLE IF NOT EXISTS ratings (
    id SERIAL PRIMARY KEY,
    official_id INTEGER REFERENCES officials(id),
    tag_id INTEGER REFERENCES tags(id),
    user_nullifier VARCHAR(255) NOT NULL,
    municipality_code INTEGER,
    state_code INTEGER,
    country_code VARCHAR(3) DEFAULT 'MX',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    CONSTRAINT unique_rating_per_user_per_official UNIQUE (official_id, user_nullifier)
);

-- Create scraped_documents table
CREATE TABLE IF NOT EXISTS scraped_documents (
    id SERIAL PRIMARY KEY,
    title VARCHAR(500) NOT NULL,
    url TEXT UNIQUE NOT NULL,
    content TEXT,
    document_type VARCHAR(100),
    source VARCHAR(255) NOT NULL,
    country_code VARCHAR(3) DEFAULT 'MX',
    municipality_code INTEGER,
    state_code INTEGER,
    publication_date TIMESTAMP,
    scraped_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    ipfs_hash VARCHAR(255),
    processed BOOLEAN DEFAULT FALSE
);

-- Create ai_analysis table
CREATE TABLE IF NOT EXISTS ai_analysis (
    id SERIAL PRIMARY KEY,
    document_id INTEGER REFERENCES scraped_documents(id),
    proposal_id INTEGER REFERENCES proposals(id),
    analysis_type VARCHAR(100) NOT NULL, -- 'impact', 'fairness', 'summary'
    analysis_result JSONB NOT NULL,
    confidence_score DECIMAL(3,2),
    model_version VARCHAR(100),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Create indexes for performance
CREATE INDEX IF NOT EXISTS idx_proposals_status ON proposals(status);

-- Create functions for updating timestamps
CREATE OR REPLACE FUNCTION update_updated_at_column()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = CURRENT_TIMESTAMP;
    RETURN NEW;
END;
$$ language 'plpgsql';

-- Create triggers for updating timestamps
CREATE TRIGGER update_users_updated_at BEFORE UPDATE ON users FOR EACH ROW EXECUTE FUNCTION update_updated_at_column();
CREATE TRIGGER update_proposals_updated_at BEFORE UPDATE ON proposals FOR EACH ROW EXECUTE FUNCTION update_updated_at_column();
CREATE TRIGGER update_officials_updated_at BEFORE UPDATE ON officials FOR EACH ROW EXECUTE FUNCTION update_updated_at_column();
